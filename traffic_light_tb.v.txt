`timescale 1ns/1ps

module traffic_light_tb;


reg clk;
reg reset;


wire red;
wire yellow;
wire green;


traffic_light uut (
    .clk(clk),
    .reset(reset),
    .red(red),
    .yellow(yellow),
    .green(green)
);

always #5 clk = ~clk;

initial
begin
    $dumpfile("dump.vcd");
    $dumpvars(0, traffic_light_tb);
end


initial
begin
    clk = 0;
    reset = 1;

    #10;
    reset = 0;

    
    #250;

    $finish;
end

initial
begin
    
    $display("Time\tGreen\tYellow\tRed");
    

    $monitor("%0t\t%b\t%b\t%b",
             $time, green, yellow, red);
end

endmodule
